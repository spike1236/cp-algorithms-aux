    consteval auto make_byte_tables() {
        std::array<std::array<uint64_t, 1 << 8>, 8> T{};
        for (int pos = 0; pos < 8; pos++) {
            for (int col = 0; col < 8; col++) {
                for (int mask = 0; mask < (1 << col); mask++) {
                    T[pos][mask | (1 << col)] = T[pos][mask] ^ COLS[pos * 8 + col];
                }
            }
        }
        return T;
    }
